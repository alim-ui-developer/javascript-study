3. 다음 코드를 Destructoring을 이용하여 해결하시오.

function calculate(obj){    
  // 함수 안을 바꾸시오
  return obj.a+obj.b+obj.c
}

calculate({a:1, b:2, c:3})

정답

const calculate = (obj) => {
  let { a, b, c } = obj;
  return a + b + c;
}

calculate({a:1, b:2, c:3})

function sumNumber() {
  //여기를 바꾸시오
  return addNumber(1)(2)(3);

  function addNumber(a) {
    return function (b) {
      return function (c) {
        return a + b + c;
      };
    };
  }
}

  const addNumber = (a) => (b) => (c) => a + b + c;
